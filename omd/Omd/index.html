A markdown parser in OCaml

The document model

The following types define the AST representing Omd's document model.

type 'attr inline =
  1. | Concat of 'attr * 'attr inline list
  2. | Text of 'attr * string
  3. | Emph of 'attr * 'attr inline
  4. | Strong of 'attr * 'attr inline
  5. | Code of 'attr * string
  6. | Hard_break of 'attr
  7. | Soft_break of 'attr
  8. | Link of 'attr * 'attr link
  9. | Image of 'attr * 'attr link
  10. | Html of 'attr * string
and 'attr link = {
  1. label : 'attr inline;
  2. destination : string;
  3. title : string option;
}
type list_type =
  1. | Ordered of int * char
  2. | Bullet of char
type list_spacing =
  1. | Loose
  2. | Tight
type cell_alignment =
  1. | Default
  2. | Left
  3. | Centre
  4. | Right
type !'attr def_elt = {
  1. term : 'attr Omd__.Ast_inline.inline;
  2. defs : 'attr Omd__.Ast_inline.inline list;
}
type !'attr block =
  1. | Blockquote of 'attr * 'attr block list
  2. | Thematic_break of 'attr
  3. | Code_block of 'attr * string * string
  4. | Html_block of 'attr * string
  5. | Definition_list of 'attr * 'attr def_elt list
type attributes = (string * string) list
type doc = attributes block list
